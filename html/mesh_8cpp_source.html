218 TriangleData Mesh::triangle( const unsigned int id ) const
219 {
220  unsigned int a, b, c;
221  if(m_indices.size() > 0)
222  {
223  assert((size_t) id*3+2 < m_indices.size());
224  a= m_indices[id*3];
225  b= m_indices[id*3 +1];
226  c= m_indices[id*3 +2];
227  }
228  else
229  {
230  assert((size_t) id*3+2 < m_positions.size());
231  a= id*3;
232  b= id*3 +1;
233  c= id*3 +2;
234  }
235 
236  TriangleData triangle;
237  triangle.a= m_positions[a];
238  triangle.b= m_positions[b];
239  triangle.c= m_positions[c];
240 
241  if(m_normals.size() == m_positions.size())
242  {
243  triangle.na= m_normals[a];
244  triangle.nb= m_normals[b];
245  triangle.nc= m_normals[c];
246  }
247  else
248  {
249  // calculer la normale geometrique
250  Vector ab= Point(m_positions[b]) - Point(m_positions[a]);
251  Vector ac= Point(m_positions[c]) - Point(m_positions[a]);
252  Vector n= normalize(cross(ab, ac));
253  triangle.na= vec3(n);
254  triangle.nb= vec3(n);
255  triangle.nc= vec3(n);
256  }
257 
258  if(m_texcoords.size() == m_positions.size())
259  {
260  triangle.ta= m_texcoords[a];
261  triangle.tb= m_texcoords[b];
262  triangle.tc= m_texcoords[c];
263  }
264  else
265  {
266  // coordonnees barycentriques des sommets, convention p(u, v)= w*a + u*b + v*c, avec w= 1 - u -v
267  triangle.ta= vec2(0, 0); // w= 1
268  triangle.tb= vec2(1, 0); // w= 0
269  triangle.tc= vec2(0, 1); // w= 0
270  }
271 
272  return triangle;
273 }

291 GLuint Mesh::create_buffers( const bool use_texcoord, const bool use_normal, const bool use_color )
292 {
293  if(m_positions.size() == 0)
294  return 0;
295 
296 #if 1
297  if(m_texcoords.size() > 0 && m_texcoords.size() < m_positions.size() && use_texcoord)
298  printf("[oops] mesh: invalid texcoords array...\n");
299  if(m_normals.size() > 0 && m_normals.size() < m_positions.size() && use_normal)
300  printf("[oops] mesh: invalid normals array...\n");
301  if(m_colors.size() > 0 && m_colors.size() < m_positions.size() && use_color)
302  printf("[oops] mesh: invalid colors array...\n");
303 #endif
304 
305  glGenVertexArrays(1, &m_vao);
306  glBindVertexArray(m_vao);
307 
308  // determine la taille du buffer pour stocker tous les attributs et les indices
309  size_t size= vertex_buffer_size() + texcoord_buffer_size() + normal_buffer_size() + color_buffer_size();
310  // allouer le buffer
311  glGenBuffers(1, &m_buffer);
312  glBindBuffer(GL_ARRAY_BUFFER, m_buffer);
313  glBufferData(GL_ARRAY_BUFFER, size, nullptr, GL_STATIC_DRAW);
314 
315  // transferer les attributs et configurer le format de sommet (vao)
316  size_t offset= 0;
317  size= vertex_buffer_size();
318  glBufferSubData(GL_ARRAY_BUFFER, offset, size, vertex_buffer());
319  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, (const void *) offset);
320  glEnableVertexAttribArray(0);
321 
322  if(m_texcoords.size() == m_positions.size() && use_texcoord)
323  {
324  offset= offset + size;
325  size= texcoord_buffer_size();
326  glBufferSubData(GL_ARRAY_BUFFER, offset, size, texcoord_buffer());
327  glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 0, (const void *) offset);
328  glEnableVertexAttribArray(1);
329  }
330 
331  if(m_normals.size() == m_positions.size() && use_normal)
332  {
333  offset= offset + size;
334  size= normal_buffer_size();
335  glBufferSubData(GL_ARRAY_BUFFER, offset, size, normal_buffer());
336  glVertexAttribPointer(2, 3, GL_FLOAT, GL_FALSE, 0, (const void *) offset);
337  glEnableVertexAttribArray(2);
338  }
339 
340  if(m_colors.size() == m_positions.size() && use_color)
341  {
342  offset= offset + size;
343  size= color_buffer_size();
344  glBufferSubData(GL_ARRAY_BUFFER, offset, size, color_buffer());
345  glVertexAttribPointer(3, 4, GL_FLOAT, GL_FALSE, 0, (const void *) offset);
346  glEnableVertexAttribArray(3);
347  }
348 
349  // allouer l'index buffer
350  if(index_buffer_size())
351  {
352  glGenBuffers(1, &m_index_buffer);
353  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, m_index_buffer);
354  glBufferData(GL_ELEMENT_ARRAY_BUFFER, index_buffer_size(), index_buffer(), GL_STATIC_DRAW);
355  }
356 
357  m_update_buffers= false;
358 
359  glBindVertexArray(0);
360  glBindBuffer(GL_ARRAY_BUFFER, 0);
361  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
362  return m_vao;
363 }

429 void Mesh::draw( const Transform& model, const Transform& view, const Transform& projection,
430  const bool use_light, const Point& light, const Color& light_color,
431  const bool use_texture, const GLuint texture,
432  const bool use_alpha_test, const float alpha_min )
433 {
434  bool use_texcoord= (m_texcoords.size() == m_positions.size() && texture > 0);
435  bool use_normal= (m_normals.size() == m_positions.size());
436  bool use_color= (m_colors.size() == m_positions.size());
437 
438  // etape 1 : construit le program en fonction des attributs du mesh et des options choisies
439  unsigned int key= 0;
440  if(use_texcoord) key= key | 1;
441  if(use_normal) key= key | 2;
442  if(use_color) key= key | 4;
443  if(use_texture) key= key | 8;
444  if(use_light) key= key | 16;
445  if(use_alpha_test) key= key | 32;
446 
447  if(m_state != key)
448  // recherche un shader deja compile pour ce type de draw
449  m_program= m_state_map[key];
450 
451  if(m_program == 0)
452  {
453  // pas de shader pour ce type de draw
454  create_program(use_texcoord, use_normal, use_color, use_light, use_alpha_test);
455  program_print_errors(m_program);
456 
457  // conserver le shader
458  m_state_map[key]= m_program;
459  }
460 
461  // conserve la config du shader selectionne.
462  assert(m_program != 0);
463  m_state= key;
464 
465  glUseProgram(m_program);
466  program_uniform(m_program, "mesh_color", default_color());
467 
468  Transform mv= view * model;
469  Transform mvp= projection * mv;
470 
471  program_uniform(m_program, "mvpMatrix", mvp);
472  program_uniform(m_program, "mvMatrix", mv);
473  if(use_normal)
474  program_uniform(m_program, "normalMatrix", mv.normal()); // transforme les normales dans le repere camera.
475 
476  // utiliser une texture, elle ne sera visible que si le mesh a des texcoords...
477  if(texture && use_texcoord && use_texture)
478  program_use_texture(m_program, "diffuse_color", 0, texture);
479 
480  if(use_light)
481  {
482  program_uniform(m_program, "light", view(light)); // transforme la position de la source dans le repere camera, comme les normales
483  program_uniform(m_program, "light_color", light_color);
484  }
485 
486  if(use_alpha_test)
487  program_uniform(m_program, "alpha_min", alpha_min);
488 
489  // etape 2 : cree les buffers et le vao
490  if(m_vao == 0)
491  create_buffers(true, true, true);
492 
493  assert(m_vao != 0);
494  if(m_update_buffers)
495  update_buffers(true, true, true);
496 
497  glBindVertexArray(m_vao);
498 
499  // etape 3 : dessiner
500  if(m_indices.size() > 0)
501  glDrawElements(m_primitives, (GLsizei) m_indices.size(), GL_UNSIGNED_INT, 0);
502  else
503  glDrawArrays(m_primitives, 0, (GLsizei) m_positions.size());
504 }
