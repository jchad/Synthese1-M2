149 int reload_program( GLuint program, const char *filename, const char *definitions )
150 {
151  if(program == 0)
152  return -1;
153 
154  // supprime les shaders attaches au program
155  int shaders_max= 0;
156  glGetProgramiv(program, GL_ATTACHED_SHADERS, &shaders_max);
157  if(shaders_max > 0)
158  {
159  std::vector<GLuint> shaders(shaders_max, 0);
160  glGetAttachedShaders(program, shaders_max, NULL, &shaders.front());
161  for (int i = 0; i < shaders_max; i++)
162  {
163  glDetachShader(program, shaders[i]);
164  glDeleteShader(shaders[i]);
165  }
166  }
167 
168 #ifdef GL_VERSION_4_3
169  glObjectLabel(GL_PROGRAM, program, -1, filename);
170 #endif
171 
172  // prepare les sources
173  std::string common_source= read(filename);
174  for(int i = 0; i < shader_keys_max; i++)
175  {
176  if(common_source.find(shader_keys[i]) != std::string::npos)
177  {
178  // cree et compile les shaders detectes dans le source
179  std::string source= prepare_source(common_source, std::string(definitions).append("#define ").append(shader_keys[i]).append("\n"));
180  GLuint shader= compile_shader(program, shader_types[i], source);
181  if(shader == 0)
182  printf("[error] compiling %s...\n%s\n", shader_string(shader_types[i]), definitions);
183  }
184  }
185 
186  // linke les shaders
187  glLinkProgram(program);
188 
189  // verifie les erreurs
190  GLint status;
191  glGetProgramiv(program, GL_LINK_STATUS, &status);
192  if(status == GL_FALSE)
193  {
194  printf("[error] linking program %u '%s'...\n", program, filename);
195  return -1;
196  }
197 
198  // pour etre coherent avec les autres fonctions de creation, active l'objet gl qui vient d'etre cree.
199  glUseProgram(program);
200  return 0;
201 }

238 void print_line( std::string& errors, const char *source, const int begin_id, const int line_id )
239 {
240  int line= 0;
241  char last= '\n';
242  for(unsigned int i= 0; source[i] != 0; i++)

248  {
249  line++;

252  errors.append(" ");
253  errors.push_back('0' + (line / 1000) % 10);

262  {
263  if(source[i] == '\t')
264  errors.append(" ");

266  errors.push_back(source[i]);
267  }
268  last= source[i];
269  }

273 int print_errors( std::string& errors, const char *log, const char *source )
274 {
275  printf("[error log]\n%s\n", log);
276 
277  int first_error= INT_MAX;
278  int last_string= -1;
279  int last_line= -1;
280  for(int i= 0; log[i] != 0; i++)
281  {
282  // recupere la ligne assiciee a l'erreur
283  int string_id= 0, line_id= 0, position= 0;
284  if(sscanf(&log[i], "%d ( %d ) : %n", &string_id, &line_id, &position) == 2 // nvidia syntax
285  || sscanf(&log[i], "%d : %d (%*d) : %n", &string_id, &line_id, &position) == 2 // mesa syntax
286  || sscanf(&log[i], "ERROR : %d : %d : %n", &string_id, &line_id, &position) == 2 // ati syntax
287  || sscanf(&log[i], "WARNING : %d : %d : %n", &string_id, &line_id, &position) == 2) // ati syntax
288  {
289  if(string_id != last_string || line_id != last_line)
290  {
291  // conserve la premiere erreur
292  first_error= std::min(first_error, line_id);
293 
294  // extrait la ligne du source...
295  errors.append("\n");
296  print_line(errors, source, last_line +1, line_id);
297  errors.append("\n");
298  }
299  }
300  // et affiche l'erreur associee...
301  for(i+= position; log[i] != 0; i++)
302  {
303  errors.push_back(log[i]);
304  if(log[i] == '\n')
305  break;
306  }
307 
308  last_string= string_id;
309  last_line= line_id;
310  }
311  errors.append("\n");
312  print_line(errors, source, last_line +1, 1000);
313  errors.append("\n");
314 
315  return first_error;
316 }

318 int program_format_errors( const GLuint program, std::string& errors )
319 {
320  errors.clear();
321 
322  if(program == 0)
323  {
324  errors.append("[error] no program...\n");
325  return -1;
326  }
327 
328  GLint status;
329  glGetProgramiv(program, GL_LINK_STATUS, &status);
330  if(status == GL_TRUE)
331  return 0;
332 
333  int first_error= INT_MAX;
334  // recupere les shaders
335  int shaders_max= 0;
336  glGetProgramiv(program, GL_ATTACHED_SHADERS, &shaders_max);
337  if(shaders_max == 0)
338  {
339  errors.append("[error] no shaders...\n");
340  return 0;
341  }
342 
343  std::vector<GLuint> shaders(shaders_max, 0);
344  glGetAttachedShaders(program, shaders_max, NULL, &shaders.front());
345  for(int i= 0; i < shaders_max; i++)
346  {
347  GLint value;
348  glGetShaderiv(shaders[i], GL_COMPILE_STATUS, &value);
349  if(value == GL_FALSE)
350  {
351  // recupere les erreurs de compilation des shaders
352  glGetShaderiv(shaders[i], GL_INFO_LOG_LENGTH, &value);
353  std::vector<char>log(value+1, 0);
354  glGetShaderInfoLog(shaders[i], (GLsizei) log.size(), NULL, &log.front());
355 
356  // recupere le source
357  glGetShaderiv(shaders[i], GL_SHADER_SOURCE_LENGTH, &value);
358  std::vector<char> source(value+1, 0);
359  glGetShaderSource(shaders[i], (GLsizei) source.size(), NULL, &source.front());
360 
361  glGetShaderiv(shaders[i], GL_SHADER_TYPE, &value);
362  errors.append("[error] compiling ").append(shader_string(value)).append("...\n");
363 
364  // formatte les erreurs
365  int last_error= print_errors(errors, &log.front(), &source.front());
366  first_error= std::min(first_error, last_error);
367  }
368  }
369 
370  // recupere les erreurs de link du program
371  {
372  GLint value= 0;
373  glGetProgramiv(program, GL_INFO_LOG_LENGTH, &value);
374 
375  std::vector<char>log(value+1, 0);
376  glGetProgramInfoLog(program, (GLsizei) log.size(), NULL, &log.front());
377 
378  errors.append("[error] linking program...\n").append(log.begin(), log.end());
379  }
380 
381  return first_error;
382 }
