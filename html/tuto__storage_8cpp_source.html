118 
119  // nombre de variables declarees
120  GLint variable_count= 0;
121  {

123  glGetProgramResourceiv(program, GL_SHADER_STORAGE_BLOCK, i, 1, prop, 1, NULL, &variable_count);
124  }

128  {
