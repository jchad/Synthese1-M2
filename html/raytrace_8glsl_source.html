33 bool intersect( const Triangle triangle, const vec3 o, const vec3 d, const float tmax, out float rt, out float ru, out float rv )
34 {
35  vec3 pvec= cross(d, triangle.ac);
36  float det= dot(triangle.ab, pvec);
37  float inv_det= 1.0f / det;
38 
39  vec3 tvec= o - triangle.a;
40  float u= dot(tvec, pvec) * inv_det;
41  vec3 qvec= cross(tvec, triangle.ab);
42  float v= dot(d, qvec) * inv_det;
43 
44  /* calculate t, ray intersects triangle */
45  rt= dot(triangle.ac, qvec) * inv_det;
46  ru= u;
47  rv= v;
48 
49  // ne renvoie vrai que si l'intersection est valide :
50  // interieur du triangle, 0 < u < 1, 0 < v < 1, 0 < u+v < 1
51  if(any(greaterThan(vec3(u, v, u+v), vec3(1, 1, 1))) || any(lessThan(vec2(u, v), vec2(0, 0))))
52  return false;
53  // comprise entre 0 et tmax du rayon
54  return (rt < tmax && rt > 0);
55 }
