49 void Gamepads::update( )
50 {
51  for(size_t p= 0; p < m_pads.size(); p++)
52  {
53  if(SDL_GameControllerGetAttached(m_pads[p].m_pad) == SDL_FALSE)
54  {
55  // le pad est debranche...
56  for(int i= 0; i < SDL_CONTROLLER_BUTTON_MAX; i++)
57  m_pads[p].m_buttons[i]= 0;
58  for(int i= 0; i < SDL_CONTROLLER_AXIS_MAX; i++)
59  m_pads[p].m_axis[i]= 0;
60 
61  SDL_GameControllerClose(m_pads[p].m_pad);
62  m_pads[p].m_pad= NULL;
63  }
64 
65  SDL_GameController *pad= m_pads[p].m_pad;
66  if(pad == NULL)
67  continue;
68 
69  // boutons
70  for(int i= 0; i < SDL_CONTROLLER_BUTTON_MAX; i++)
71  if(SDL_GameControllerGetButton(pad, (SDL_GameControllerButton) i) == SDL_PRESSED)
72  m_pads[p].m_buttons[i]= 1;
73  else
74  m_pads[p].m_buttons[i]= 0;
75 
76  // axes
77  for(int i= 0; i < SDL_CONTROLLER_AXIS_MAX; i++)
78  {
79  int value= SDL_GameControllerGetAxis(pad, (SDL_GameControllerAxis) i);
80  if(value > -6000 && value < 6000)
81  // dead zone...
82  value= 0;
83 
84  m_pads[p].m_axis[i]= (float) value / 32768.f;
85  }
86  }
