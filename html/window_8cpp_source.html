158 int events( Window window )
159 {
160  event_count= 0;
161 
162  // proportions de la fenetre
163  SDL_GetWindowSize(window, &width, &height);
164  aspect= float(width) / float(height);
165 
166  // gestion des evenements
167  SDL_Event event;
168  while(SDL_PollEvent(&event))
169  {
170  event_count++;
171 
172  switch(event.type)
173  {
174  case SDL_WINDOWEVENT:
175  // redimensionner la fenetre...
176  if(event.window.event == SDL_WINDOWEVENT_RESIZED)
177  {
178  // conserve les proportions de la fenetre
179  //~ width= event.window.data1;
180  //~ height= event.window.data2;
181  width= std::floor(event.window.data2 * aspect);
182  height= event.window.data2;
183  SDL_SetWindowSize(window, width, height);
184 
185  // ... et le viewport opengl
186  glViewport(0, 0, width, height);
187  }
188  break;
189 
190  case SDL_DROPFILE:
191  last_drop.assign(event.drop.file);
192  SDL_free(event.drop.file);
193  break;
194 
195  case SDL_TEXTINPUT:
196  // conserver le dernier caractere
197  last_text= event.text;
198  break;
199 
200  case SDL_KEYDOWN:
201  // modifier l'etat du clavier
202  if((size_t) event.key.keysym.scancode < key_states.size())
203  {
204  key_states[event.key.keysym.scancode]= 1;
205  last_key= event.key; // conserver le dernier evenement
206  }
207 
208  // fermer l'application
209  if(event.key.keysym.sym == SDLK_ESCAPE)
210  stop= 1;
211  break;
212 
213  case SDL_KEYUP:
214  // modifier l'etat du clavier
215  if((size_t) event.key.keysym.scancode < key_states.size())
216  {
217  key_states[event.key.keysym.scancode]= 0;
218  last_key= event.key; // conserver le dernier evenement
219  }
220  break;
221 
222  case SDL_MOUSEBUTTONDOWN:
223  case SDL_MOUSEBUTTONUP:
224  last_button= event.button;
225  break;
226 
227  case SDL_MOUSEWHEEL:
228  last_wheel= event.wheel;
229  break;
230 
231  case SDL_QUIT:
232  stop= 1; // fermer l'application
233  break;
234  }
235  }
236 
237  return 1 - stop;
238 }
