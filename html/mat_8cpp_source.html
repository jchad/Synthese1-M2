48 Point Transform::operator() ( const Point& p ) const
49 {
50  float x= p.x;
51  float y= p.y;
52  float z= p.z;
53 
54  float xt= m[0][0] * x + m[0][1] * y + m[0][2] * z + m[0][3]; // dot(vec4(m[0]), vec4(p, 1))
55  float yt= m[1][0] * x + m[1][1] * y + m[1][2] * z + m[1][3]; // dot(vec4(m[1]), vec4(p, 1))
56  float zt= m[2][0] * x + m[2][1] * y + m[2][2] * z + m[2][3]; // dot(vec4(m[2]), vec4(p, 1))
57  float wt= m[3][0] * x + m[3][1] * y + m[3][2] * z + m[3][3]; // dot(vec4(m[3]), vec4(p, 1))
58 
59  assert(wt != 0);
60  float w= 1.f / wt;
61  if(wt == 1.f)
62  return Point(xt, yt, zt);
63  else
64  return Point(xt*w, yt*w, zt*w);
65 }

281 Transform Transform::inverse( ) const
282 {
283  Transform minv= *this;
284 
285  int indxc[4], indxr[4];
286  int ipiv[4] = { 0, 0, 0, 0 };
287 
288  for (int i = 0; i < 4; i++) {
289  int irow = -1, icol = -1;
290  float big = 0.f;
291 
292  // Choose pivot
293  for (int j = 0; j < 4; j++) {
294  if (ipiv[j] != 1) {
295  for (int k = 0; k < 4; k++) {
296  if (ipiv[k] == 0) {
297  if (fabsf(minv.m[j][k]) >= big) {
298  big = std::abs(minv.m[j][k]);
299  irow = j;
300  icol = k;
301  }
302  }
303  else if (ipiv[k] > 1)
304  printf("singular matrix in make_inverse()\n");
305  }
306  }
307  }
308 
309  assert(irow >= 0 && irow < 4);
310  assert(icol >= 0 && icol < 4);
311 
312  ++ipiv[icol];
313  // Swap rows _irow_ and _icol_ for pivot
314  if (irow != icol) {
315  for (int k = 0; k < 4; ++k)
316  std::swap(minv.m[irow][k], minv.m[icol][k]);
317  }
318 
319  indxr[i] = irow;
320  indxc[i] = icol;
321  if (minv.m[icol][icol] == 0.)
322  printf("singular matrix in make_inverse()\n");
323 
324  // Set $m[icol][icol]$ to one by scaling row _icol_ appropriately
325  float pivinv = 1.f / minv.m[icol][icol];
326  minv.m[icol][icol] = 1.f;
327  for (int j = 0; j < 4; j++)
328  minv.m[icol][j] *= pivinv;
329 
330  // Subtract this row from others to zero out their columns
331  for (int j = 0; j < 4; j++) {
332  if (j != icol) {
333  float save = minv.m[j][icol];
334  minv.m[j][icol] = 0;
335  for (int k = 0; k < 4; k++)
336  minv.m[j][k] -= minv.m[icol][k]*save;
337  }
338  }
339  }
340 
341  // Swap columns to reflect permutation
342  for (int j = 3; j >= 0; j--) {
343  if (indxr[j] != indxc[j]) {
344  for (int k = 0; k < 4; k++)
345  std::swap(minv.m[k][indxr[j]], minv.m[k][indxc[j]]);
346  }
347  }
348 
349  return minv;
350 }
